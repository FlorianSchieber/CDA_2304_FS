VARIABLES
	Chaîne : phrase
	Entier : indiceGauche
	Entier : indiceDroit
	Booléen : estUnPalindrome

DEBUT PROGRAMME
	Faire
		Faire
			Ecrire "Veuillez saisir une phrase terminée par un point : "
			Lire phrase
			Si phrase = "" ou phrase = "."
				Ecrire "Votre phrase est VIDE !"
			Fin Si
		Répéter Tant Que phrase = "" ou phrase = "."
		Si phrase[|phrase| - 1] ≠ "."
			Ecrire "Vous n'avez pas mis de POINT à la fin de votre phrase !"
		Fin Si
	Répéter Tant Que phrase[|phrase| - 1] ≠ "."
	indiceGauche <- 0
	indiceDroit <- |phrase| - 2
	estUnPalindrmome <- VRAI
	Tant Que indiceGauche < indiceDroit et estUnPalindrome
		Si phrase[indiceGauche] = " " // Suppression des espaces gauches
			indiceGauche <- indiceGauche + 1
		Sinon si phrase[indiceDroit] = " " // Suppression des espaces droites
			indiceDroit <- indiceDroit - 1
		Sinon 
			Si phrase[indiceGauche] ≠ phrase[indiceDroit]
				estUnPalindrome <- FAUX
			Fin Si
			indiceGauche <- indiceGauche + 1
			indiceDroit <- indiceDroit - 1
		Fin Si
	Fin Tant Que
	Si estUnPalindrome
		Ecrire "La phrase est un palindrome"
	Sinon
		Ecrire "La phrase n'est pas un palindrome"
FIN PROGRAMME
