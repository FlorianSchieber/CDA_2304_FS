VARIABLES
	Entier : saisie
	Entier : nombreMystere
	Entier : minimumFouchette
	Entier : maximumFouchette
	Entier : essais
	Booléen : aTrouve

CONSTANTES
	Entier : DEPART_MINIMUM = 0
	Entier : DEPART_MAXIMUM = 100

DEBUT PROGRAMME
	nombreMystere <- RANDOM(DEPART_MINIMUM, DEPART_MAXIMUM)
	minimumFourchette <- DEPART_MINIMUM
	maximumFourchette <- DEPART_MINIMUM
	essais <- 0
	aTrouve <- FAUX
	Tant Que non aTrouve
		Faire
			Ecrire "Veuillez saisir un nombre entre", minimumFourchette, " et ", maximumFouchette, " : "
			Lire saisie
		Répéter Tant Que saisie < minimumFourchette ou saisie > maximumFourchette
		essais <- essais + 1
		Si saisie < nombreMystere
			Ecrire "Trop petit !"
			minimumFourchette <- saisie + 1		
		Sinon si saisie > nombreMystere
			Ecrire "Trop grand !"
			maximumFourchette <- saisie - 1
		Sinon
			Ecrire "Bravo, vous avez trouvé en ", essais, " essais."
			aTrouve <- VRAI
		Fin Si
	Fin Tant Que
FIN PROGRAMME
