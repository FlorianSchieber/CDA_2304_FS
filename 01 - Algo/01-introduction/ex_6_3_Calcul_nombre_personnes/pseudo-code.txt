VARIABLES
	Entier : personne
	Entier : jeunes
	Entier : ages[PERSONNES]

CONSTANTES
	Entier : AGE_VIEUX <- 20
	Entier : PERSONNES <- 20
	

DEBUT PROGRAMME
	jeunes <- 0
	Pour personne De 0 à PERSONNES - 1
		Ecrire "Veuillez entrer l'âge de la personne n°", personne + 1
		Lire ages[personne]
		Si ages[personne] < AGE_VIEUX
			jeunes <- jeunes + 1
		Fin Si
	Fin Pour
	Si jeunes = 0
		Ecrire "Toutes les personnes ont moins de ", AGE_VIEUX, " ans."
	Sinon Si jeunes = PERSONNES
		Ecrire "Toutes les personnes ont ", AGE_VIEUX, " ans ou plus."
	Sinon
		Ecrire "Il y a ", jeunes, " personnes de moins de ", AGE_VIEUX, " ans, et ", PERSONNES - jeunes, " autres personnes."
	Fin Si
FIN PROGRAMME
