VARIABLES
	ENTIER : N
	ENTIER : diviseur
	BOOLÉEN : estCompose

DEBUT PROGRAMME
	ÉCRIRE "Veuillez entrer un nombre : "
	LIRE N
	SI N > 1
		estCompose <-- FAUX
		diviseur <-- 2
		TANT QUE NON estCompose ET diviseur < N
			SI N MOD diviseur = 0 // Si on trouve un diviseur
				estCompose <-- VRAI
			FIN SI
			diviseur <-- diviseur + 1 
		FIN TANT QUE
		SI estCompose
			ÉCRIRE N + " est composé."
		Sinon
			ÉCRIRE N + " est premier."
		FIN SI
	SINON
		ÉCRIRE "Par définition, " + N + " n'est pas premier."
FIN PROGRAMME
